/*
Write a program to find the factorial value of any number
entered through the keyboard.
*/
#include <stdio.h>
int main(){
	int num = 5;
	int rs;
	if (num == 1)
	{
		return num;
	}
	else if (num == 2){
		return 2;
	}
	else{
		rs= factorial(num);
		return rs;
	}
}
int factorial(int number){
	int result = number * factorial(number -1);
	return result;
}
